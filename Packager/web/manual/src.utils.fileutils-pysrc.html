 88 -def dirEntries(dir_name, subdir, extension): 

 89      """

 93      Additional argument, if any, is file extension to match filenames. 
 94   
 95      """ 
 96      fileList = [] 
 97      for file in os.listdir(dir_name): 
 98          dirfile = os.path.join(dir_name, file) 
 99          if os.path.isfile(dirfile): 
100              if not extension: 
101                  fileList.append(dirfile) 
102              else: 
103                  if dirfile.lower().endswith(extension.lower()): 
104                      fileList.append(dirfile) 
105          # recursively access file names in subdirectories 
106          elif os.path.isdir(dirfile) and subdir: 
107              fileList.extend(dirEntries(dirfile, subdir, extension)) 
108      return fileList
